class Produto{
    int codigo_barras;
    String descricao;
    double preco;
    
    Produto(int codigo_barras, String descricao, double preco){
    this.codigo_barras = codigo_barras;
    this.descricao = descricao;
    this.preco = preco;
    }
    void exibir(){
        System.out.println("Código de barras: " + codigo_barras + 
        "\nDescrição: " + descricao + 
        "\nPreço: " + preco);
        System.out.println("-----------------------");
    }
}
